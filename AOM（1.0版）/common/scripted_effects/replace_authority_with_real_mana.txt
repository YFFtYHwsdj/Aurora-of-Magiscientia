set_countries_of_holy_church = {
	set_global_variable = {
		name = total_population_of_holy_church
		value = 0
	}
	every_country = {
		if = {
			limit = {
				religion = rel:holy_church
			}
			if = {
				limit = {
					NOT = {
						this = ROOT
					}
				}

				change_global_variable = {
					name = total_population_of_holy_church
					add = this.total_population
				}
			}
			if = {
				limit = {
					total_population < 5000000
				}
				add_modifier = {
					name = mana_50
					months = 1
