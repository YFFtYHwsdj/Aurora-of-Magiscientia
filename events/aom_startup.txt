namespace = aom_startup

# change aristoric pops in old empires into acane
aom_startup.1 = {
	type = country_event
	hidden = yes

	immediate = {
		every_scope_pop = {
			limit = {
				is_pop_type = aristocrats
			}
			change_pop_religion = {
				target = rel:linegism
				value = 1
			}
		}
	}
}

# change magic pops without magical relighion into magical
aom_startup.2 = {
	type = country_event
	hidden = yes

	immediate = {
		every_scope_pop = {
			limit = {
				NOT = {
					religion = {
						has_discrimination_trait = magic
					}
				}
				OR  = {
					is_pop_type = grand_magicians
					is_pop_type = magicians
				}
			}
			change_pop_religion = {
				target = rel:arcane
				value = 1
			}
		}
	}
}
