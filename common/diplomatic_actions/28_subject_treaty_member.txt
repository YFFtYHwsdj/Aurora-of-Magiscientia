subject_treaty_member = {
	require_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = no
	
	potential = {
		c:PAP = root
		c:EMP = target_country
	}
	
	possible = {
		c:PAP = root
		c:EMP = target_country
		NOT = { has_war_with = scope:target_country }
		root = {
			NOT = {
				has_diplomatic_pact = {
					who = scope:target_country
					type = subject_treaty_member
				}
			}
		}
		scope:target_country = {
			NOT = {
				has_diplomatic_pact = {
					who = root
					type = subject_treaty_member
				}
			}
		}
	}
	
	pact = {
		cost = 100
		target_requires_approval_to_break = yes
		income_transfer = 0.3
	}
}
