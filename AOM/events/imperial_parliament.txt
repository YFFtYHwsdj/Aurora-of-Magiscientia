#创建变量
ip_poll.1 = {
	type = none
	
	title = ip_poll.1.t
	desc = ip_poll.1.d
	
	hidden = yes
	
	trigger = {
		always = yes
	}
	
	immediate = {
		set_global_variable = {
			name = pro_values
			value = 0
		}
		set_global_variable = {
			name = con_values
			value = 0
		}
		set_global_variable = {
			name = waiver_values
			value = 0
		}
	}
}
	

# 投票收集 每月收集
ip_poll.2 = {
	type = none
	
	title = ip_poll.2.t
	desc = ip_poll.2.d
	
	hidden = yes
	
	trigger = {
		always = yes
	}
	
	immediate = {
		every_country = {
			
			limit = {
				has_modifier = has_voting_right_modifier
			}
		
			if = {
				limit = {
					has_modifier = pro_modifier
				}
				change_variable = {
					name = pro_values
					add = root.prestige
				}
			}
			else_if = {
				limit = {
					has_modifier = con_modifier
				}
				change_variable = {
					name = con_values
					add = root.prestige
				}
			}
			else_if = {
				limit = {
					has_modifier = waiver_modifier
				}
				change_variable = {
					name = waiver_values
					add = root.prestige
				}
			}
			else = {
			}
			
		}
	}
}
# 投票收集 每月初始化
ip_poll.3 = {
	type = none
	
	title = ip_poll.3.t
	desc = ip_poll.3.d
	
	hidden = yes
	
	trigger = {
		always = yes
	}
	
	immediate = {
		set_global_variable = {
			name = pro_values
			value = 0
		}
		set_global_variable = {
			name = con_values
			value = 0
		}
		set_global_variable = {
			name = waiver_values
			value = 0
		}
		trigger_event = {
			id = ip_poll.2
			days = 0
			popup = no
		}
	}
}
