subject_church_vassal = {
	require_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = no
	
	potential = {
		c:PAP = root
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
			is_subject_of = root
		}
	}
	
	possible = {
		c:PAP = root
		NOT = { has_war_with = scope:target_country }
		root = {
			NOT = {
				has_diplomatic_pact = {
					who = scope:target_country
					type = subject_church_vassal
				}
			}
		}
		scope:target_country = {
			NOT = {
				has_diplomatic_pact = {
					who = root
					type = subject_church_vassal
				}
			}
		}
	}
	
	pact = {
		cost = 50
		target_requires_approval_to_break = yes
		income_transfer = 0.3
	}
}
