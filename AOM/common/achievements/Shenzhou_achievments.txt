# possible - filters out achivements on game start so we dont have to check everything all the time
# happened - checks if achivement happened

the_end_of_an_era = {
	possible = {
		exists = c:NWU
		this = c:NWU
	}
	happened = {
		exists = c:NWU
		this = c:NWU
		NOT = { exists = c:BZH }
		any_state = {
			state_region = { 
				OR = {
					is_homeland = cu:loong
					is_homeland = cu:oriental
				}
				
			} 
			owner = c:NWU
		}
		owns_entire_state_region = STATE_B14
	}
}
