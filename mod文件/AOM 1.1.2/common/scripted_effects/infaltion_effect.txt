infaltion_effect = {
    remove_modifier = hyper_inflation_modifier
    remove_modifier = inflation_modifier
    if = {
        limit = {
            root.market = {
                owner = root
            }
        }
        set_variable = {
            name = inflation
            value = market_inflation
        }
        if = {
            limit = {
                var:inflation > 0.1
                # hyper inflation
            }
            root.market = {
                every_scope_country = {
                    add_modifier = {
                        name = hyper_inflation_modifier
                        multiplier = var:inflation
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:inflation > 0
            }
            root.market = {
                every_scope_country = {
                    add_modifier = {
                        name = inflation_modifier
                        multiplier = var:inflation
                    }
                }
            }
        }
        remove_variable = inflation
    }
}
