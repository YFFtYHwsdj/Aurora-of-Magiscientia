# 跨市场贸易中心，开始在国内渗透，并且对于本地的贸易中心有发展加成
cc_trade_center_spreading = {
    every_scope_state = {
        limit = {
            has_building = building_trade_center
            market_access < 1
        }
        if = {
            limit = {
                NOT = { has_building = building_domestic_trade_center }
                any_scope_building = {
                    is_building_type = building_trade_center
                    cash_reserves_available > 5000
                }
            } 
            create_building = {
                building = "building_domestic_trade_center"
                level = 1
                reserves = 1
            }
        else_if = {
            limit = {
                any_scope_building = {
                    is_building_type = building_trade_center
                    cash_reserves_available > 10000
                any_scope_building = {
                    is_building_type = building_domestic_trade_center
                    weekly_profit > 0
                }
            }
            add_building_level = {
                building = building_domestic_trade_center
                level = 2
            }  
        }    
    } 
}

# 本地贸易中心的升级
cc_domestic_trade_center_expanding = {
    every_scope_state = {
        limit = {
            market_access < 1
            any_scope_building = {
                is_building_type = building_domestic_trade_center
                cash_reserves_available > 5000
                weekly_profit > 0
            }
        }
        save_scope_as = cc_target_state
    }
    scope:cc_target_state = {
        add_building_level = {
            building = building_domestic_trade_center
            level = 1
        }
    }
}

# 本地贸易中心在国内的渗透
cc_domestic_trade_center_spreading = {
    every_scope_state = {
        limit = {
            any_scope_building = {
                is_building_type = building_domestic_trade_center
                weekly_profit > 0
            }
        }
        random_neighbouring_state = {
            limit = {
                NOT = { has_building = building_domestic_trade_center }
                market_access < 1
                OR = {
                    owner = ROOT
                    owner = {
                        OR = {
                            has_diplomatic_pact = {
                                who = ROOT
                                type = puppet
                            } 
                            has_diplomatic_pact = {
                                who = ROOT
                                type = domain
                            }
                            has_diplomatic_pact = {
                                who = ROOT
                                type = customs_union
                            }
                        }
                    } 
                }
            }     
            create_building = {
                building = "building_domestic_trade_center"
                level = 1
                reserves = 1
            }
        }
        if = {
            limit = {
                any_scope_building = {
                    is_building_type = building_domestic_trade_center
                    cash_reserves_available > 10000
                }
            random_neighbouring_state = { 
                limit = {
                    has_building = building_domestic_trade_center
                    market_access < 1
                    OR = {
                        owner = ROOT
                        owner = {
                            OR = {
                                has_diplomatic_pact = {
                                    who = ROOT
                                    type = puppet
                                } 
                                has_diplomatic_pact = {
                                    who = ROOT
                                    type = domain
                                }
                                has_diplomatic_pact = {
                                    who = ROOT
                                    type = customs_union
                                }
                            }
                        } 
                    }
                }      
                add_building_level = {
                    building = building_domestic_trade_center
                    level = 2
                }  
            } 
        } 
    }    
}
