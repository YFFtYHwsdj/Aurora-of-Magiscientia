set_countries_of_holy_church = {
	set_global_variable = {
		name = total_population_of_holy_church
		value = 0
	}
	every_country = {
		if = {
			limit = {
				religion = rel:catholic
			}
			change_global_variable = {
				name = total_population_of_holy_church
				add = this.total_population
			}
			set_variable = {
				name = target_population
				value = this.total_population
			}
			remove_modifier = mana_holy_church
			add_modifier = {
				name = mana_holy_church
				multiplier = aom_mana_multiplier
			}
		}
	}
	set_variable = {
		name = target_population
		value = global_var:total_population_of_holy_church
	}
	remove_modifier = papa_mana_holy_church
	add_modifier = {
		name = papa_mana_holy_church
		multiplier = aom_mana_multiplier
	}
}
